/** * SENZANY — Centre de commandement * assets/js/admin/admin.js */ const elements = { loading: document.getElementById("adminLoading"), loggedOut: document.getElementById("adminLoggedOut"), accessDenied: document.getElementById("adminAccessDenied"), error: document.getElementById("adminError"), errorMessage: document.getElementById("adminErrorMessage"), retryButton: document.getElementById("adminRetryButton"), dashboard: document.getElementById("adminDashboard"), backendStatus: document.getElementById("adminBackendStatus"), homeView: document.getElementById("adminHomeView"), deliveriesView: document.getElementById("adminDeliveriesView"), deliveriesButton: document.querySelector('[data-admin-module="deliveries"]'), backButton: document.getElementById("backToAdminHome") }; function hideAll() { elements.loading.hidden = true; elements.loggedOut.hidden = true; elements.accessDenied.hidden = true; elements.error.hidden = true; elements.dashboard.hidden = true; } function show(view) { hideAll(); switch (view) { case "loading": elements.loading.hidden = false; break; case "loggedOut": elements.loggedOut.hidden = false; break; case "denied": elements.accessDenied.hidden = false; break; case "error": elements.error.hidden = false; break; case "dashboard": elements.dashboard.hidden = false; break; } } function showHome() { if (elements.homeView) elements.homeView.hidden = false; if (elements.deliveriesView) elements.deliveriesView.hidden = true; } function showDeliveries() { if (elements.homeView) elements.homeView.hidden = true; if (elements.deliveriesView) elements.deliveriesView.hidden = false; } async function checkAccess() { show("loading"); try { const response = await fetch("/api/steam/me", { credentials: "same-origin", cache: "no-store" }); if (!response.ok) throw new Error("Backend indisponible"); const user = await response.json(); console.log(user); const logged = user.logged || user.loggedIn || user.authenticated; if (!logged) { show("loggedOut"); return; } const staff = user.isStaff || user.staff || user.admin || user.isAdmin || user.role === "staff" || user.role === "admin"; if (!staff) { show("denied"); return; } if (elements.backendStatus) elements.backendStatus.textContent = "ONLINE"; show("dashboard"); showHome(); } catch (err) { console.error(err); if (elements.backendStatus) elements.backendStatus.textContent = "ERREUR"; if (elements.errorMessage) elements.errorMessage.textContent = err.message; show("error"); } } document.addEventListener("DOMContentLoaded", () => { if (elements.deliveriesButton) elements.deliveriesButton.addEventListener("click", showDeliveries); if (elements.backButton) elements.backButton.addEventListener("click", showHome); if (elements.retryButton) elements.retryButton.addEventListener("click", checkAccess); checkAccess(); });
